     *
     * AUTH-03/04: signIn callback is a STUB in this plan. Full
     * customer-membership enforcement + staff isStaff flag lands in 18-03.
     *
     * AUTH-06: Email is the canonical user identifier. NEVER reference token.sub
     * for permission decisions — vitest grep-test in 18-05 enforces this.
     */
    export const authOptions: NextAuthOptions = {
      providers: [
        GoogleProvider({
          clientId: process.env.GOOGLE_CLIENT_ID!,
          clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
          authorization: {
            params: {
              scope: 'openid email profile',
              access_type: 'offline',
              prompt: 'consent',
            },
          },
        }),
      ],
      callbacks: {
        async signIn({ user }) {
          // STUB — full membership rule (allow members + staff, reject
          // 0-membership users) lands in 18-03. For now, allow any
          // authenticated email through so the OAuth round-trip can be tested.
          const email = user.email ?? '';
          return Boolean(email);
        },
        async jwt({ token, account }) {
          if (account) {
            token.accessToken = account.access_token;
          }
          return token;
        },
        async session({ session, token }: { session: Session; token: JWT }) {
          (session as Session & { accessToken?: string }).accessToken = token.accessToken as string | undefined;
          return session;
        },
      },
      cookies: {
        sessionToken: {
          name: process.env.NODE_ENV === 'production'
            ? `__Host-next-auth.session-token`
            : `next-auth.session-token`,

            httpOnly: true,
            sameSite: 'lax',
            path: '/',
            secure: process.env.NODE_ENV === 'production',
            // NO domain field — host-only by omission (Pitfall 1)

        csrfToken: {
          name: process.env.NODE_ENV === 'production'
            ? `__Host-next-auth.csrf-token`

          options: {
            httpOnly: true,
            sameSite: 'lax',
            path: '/',

          },

            : `next-auth.callback-url`,
          options: {
            sameSite: 'lax',
            path: '/',
            secure: process.env.NODE_ENV === 'production',
          },
        },
      },
      session: { strategy: 'jwt' },
      pages: { signIn: '/login' },
      secret: process.env.NEXTAUTH_SECRET,
