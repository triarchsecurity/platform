    ```tsx
    import { redirect } from 'next/navigation';
    import { getCurrentUserContext } from '@myalterlego/triarch-shared/auth';
    import { getPortalSession } from '@/lib/session';

    /**
     * AUTH-07: Post-login routing decision tree.
     *
     * Unauthenticated → /login
     * Authenticated, 0 memberships → /no-memberships (empty state)
     * Authenticated, 1 membership   → /projects/{key}/releases (auto-redirect)
     * Authenticated, 2+ memberships → /projects (list)
     *
     * Staff users with 0 memberships are also routed to /no-memberships;
     * the StaffCallout banner (rendered site-wide via layout) tells them
     * to switch to admin.triarch.dev.
     */
    export default async function HomePage() {
      const session = await getPortalSession();

      if (!session?.user?.email) {
        redirect('/login');
      }

      const ctx = await getCurrentUserContext({ user: { email: session.user.email } });

      // Defense in depth: signIn callback already rejects null ctx, but
      // session strategy is jwt and ctx may be null here on transient DB error.
      // Redirect to no-memberships as the safest fallback (StaffCallout still
      // shows for staff; customer sees Contact-your-admin copy).
      if (ctx === null || ctx.memberships.length === 0) {
        redirect('/no-memberships');
      }

      if (ctx.memberships.length === 1) {
        redirect(`/projects/${ctx.memberships[0].projectKey}/releases`);
      }

      // 2+ memberships → list page
      redirect('/projects');
    }
    ```
