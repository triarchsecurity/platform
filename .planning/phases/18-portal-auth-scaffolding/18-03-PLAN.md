    import type { JWT } from 'next-auth/jwt';
    import GoogleProvider from 'next-auth/providers/google';
    import { getCurrentUserContext } from '@myalterlego/triarch-shared/auth';

    /**
     * Portal NextAuth v4 configuration.
     *
     * AUTH-01: Cookie config uses `__Host-` prefix in production with NO `domain`
     * attribute (host-only). Pitfall 1 guard against cross-subdomain leakage.
     *
     * AUTH-02: NEXTAUTH_SECRET reads from process.env, bound to
     * PORTAL_NEXTAUTH_SECRET in apphosting.yaml. Distinct from admin's secret.
     *
     * AUTH-03: signIn callback queries getCurrentUserContext for the email and
     * rejects users with 0 memberships AND not staff. Staff users with no
     * project memberships are still allowed in (they get the StaffCallout
     * banner via AUTH-04 instead of a 401).
     *
     * AUTH-04: jwt + session callbacks propagate isStaff into session.user so
     * the StaffCallout server component can render conditionally.
     *
     * AUTH-06: Email is the canonical user identifier. NEVER reference
     * token.sub for permission decisions.
     */
    export const authOptions: NextAuthOptions = {
      providers: [
        GoogleProvider({
          clientId: process.env.GOOGLE_CLIENT_ID!,
          clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
          authorization: {
            params: {
              scope: 'openid email profile',
              access_type: 'offline',
              prompt: 'consent',
            },
          },
        }),
      ],
      callbacks: {
        async signIn({ user }) {
          const email = user.email ?? '';
          if (!email) return false;

          // Real customer-membership rule (replaces 18-02's stub).
          // getCurrentUserContext queries project_members via shared package.
          // Returns null on DB error → fail closed (reject sign-in).
          const ctx = await getCurrentUserContext({ user: { email } });
          if (ctx === null) {
            console.error('[portal-auth] getCurrentUserContext returned null for', email);
            return false;
          }

          // Allow staff (even with 0 memberships — they see StaffCallout).
          // Allow customer admin/viewer (memberships > 0).
          // Reject everyone else (0 memberships AND not staff).
          if (ctx.isStaff) return true;
          if (ctx.memberships.length > 0) return true;
          return false;
        },
        async jwt({ token, account, user }) {
          // First sign-in: account is set, user has email.
          if (account && user?.email) {
            token.email = user.email;
            // Re-query getCurrentUserContext to compute isStaff for the JWT.
            const ctx = await getCurrentUserContext({ user: { email: user.email } });
            token.isStaff = ctx?.isStaff ?? false;
            token.accessToken = account.access_token;
          }
          return token;
        },
        async session({ session, token }: { session: Session; token: JWT }) {
          if (session.user) {
            session.user.isStaff = token.isStaff ?? false;
          }
          (session as Session & { accessToken?: string }).accessToken = token.accessToken as string | undefined;
          return session;
        },
      },
      cookies: {
        sessionToken: {
          name: process.env.NODE_ENV === 'production'
            ? `__Host-next-auth.session-token`
            : `next-auth.session-token`,
          options: {
            httpOnly: true,
            sameSite: 'lax',
            path: '/',
            secure: process.env.NODE_ENV === 'production',
            // NO domain field — host-only by omission (Pitfall 1)
          },
        },
        csrfToken: {
          name: process.env.NODE_ENV === 'production'
            ? `__Host-next-auth.csrf-token`
            : `next-auth.csrf-token`,
          options: {
            httpOnly: true,
            sameSite: 'lax',
            path: '/',
            secure: process.env.NODE_ENV === 'production',
          },
        },
        callbackUrl: {
          name: process.env.NODE_ENV === 'production'
            ? `__Secure-next-auth.callback-url`
            : `next-auth.callback-url`,
          options: {
            sameSite: 'lax',
            path: '/',
            secure: process.env.NODE_ENV === 'production',
          },
        },
      },
      session: { strategy: 'jwt' },
      pages: { signIn: '/login' },
      secret: process.env.NEXTAUTH_SECRET,
    };
