    ```ts
    /**
     * AUTH-06 + Pitfall 2 guard: portal source files MUST NOT reference the
     * OAuth `sub` claim. Email is the canonical user identifier — using `sub`
     * creates ghost identities because `sub` is per-OAuth-client, not
     * per-Google-account.
     *
     * This test scans src/ for any `.sub` reference and fails if found,
     * excluding this test file itself.
     */
    import { describe, it, expect } from 'vitest';
    import { execSync } from 'node:child_process';

    describe('AUTH-06: no .sub claim references in src/', () => {
      it('grep -rn "\\.sub\\b" src/ returns no matches outside this test file', () => {
        // -rn: recursive, line numbers
        // \.sub\b: literal .sub followed by word boundary (matches token.sub, profile.sub, etc.; not .substr, .substring)
        // --exclude: skip this test file itself
        // || true: grep exits 1 when no match, which would fail the bash exec; we capture stdout regardless
        const result = execSync(
          `grep -rn "\\.sub\\b" src/ --exclude="no-sub-claim.test.ts" || true`,
          { encoding: 'utf-8' }
        );

        // Filter out empty lines and any noise; we want zero hits.
        const hits = result
          .split('\n')
          .filter((line) => line.trim().length > 0);

        if (hits.length > 0) {
          throw new Error(
            `Found ${hits.length} reference(s) to .sub claim in portal src/ — use email instead:\n` +
              hits.join('\n')
          );
        }

        expect(hits.length).toBe(0);
      });
    });
    ```

    Note on regex: the pattern `\.sub\b` matches `.sub` followed by a word boundary, so it WILL match `token.sub`, `profile.sub`, `user.sub` but will NOT match `.substring`, `.substr`, `.subscribe`. This is intentional. If a future contributor uses `.subscribe` for a stream, the test stays green.

    The `--exclude` flag excludes only the test file itself by basename. If portal grows additional test fixtures referencing `.sub` for legitimate reasons, this exclusion list will need extension — for now AUTH-06 dictates zero references anywhere in src/.

    ```ts
    /**
     * AUTH-03 unit tests: signIn callback enforces customer-membership rule
     * with staff bypass.
     *
     * Allow matrix:
     *   email empty                            → false
     *   ctx === null (DB error)                → false
     *   ctx.memberships=[], isStaff=false      → false (reject 0-membership customer)
     *   ctx.memberships=[], isStaff=true       → true  (staff bypass — gets StaffCallout)
     *   ctx.memberships=[X], isStaff=false     → true  (customer admin/viewer)
     *   ctx.memberships=[X], isStaff=true      → true  (staff who is also a member)
     */
    import { describe, it, expect, vi, beforeEach } from 'vitest';

    vi.mock('@myalterlego/triarch-shared/auth', () => ({
      getCurrentUserContext: vi.fn(),
    }));

    import { authOptions } from './auth';
    import { getCurrentUserContext } from '@myalterlego/triarch-shared/auth';

    const mockGetCtx = vi.mocked(getCurrentUserContext);

    type MockUser = { email: string | null; name?: string | null; image?: string | null };

    async function callSignIn(user: MockUser) {
      const cb = authOptions.callbacks?.signIn;
      if (!cb) throw new Error('signIn callback not defined');
      // NextAuth signature: { user, account, profile, email, credentials } — we only need user
      // @ts-expect-error -- partial args for unit test
      return cb({ user });
    }

    describe('AUTH-03: signIn callback', () => {
      beforeEach(() => {
        mockGetCtx.mockReset();
      });

      it('returns false for empty email', async () => {
        const result = await callSignIn({ email: null });
        expect(result).toBe(false);
        expect(mockGetCtx).not.toHaveBeenCalled();
      });

      it('returns false when getCurrentUserContext returns null (DB error fail-closed)', async () => {
        mockGetCtx.mockResolvedValueOnce(null);
        const result = await callSignIn({ email: 'unknown@example.com' });
        expect(result).toBe(false);
      });

      it('returns false for 0-membership non-staff user', async () => {
        mockGetCtx.mockResolvedValueOnce({
          email: 'random@external.com',
          isStaff: false,
          memberships: [],
        });
        const result = await callSignIn({ email: 'random@external.com' });
        expect(result).toBe(false);
      });

      it('returns true for staff user with 0 memberships (StaffCallout bypass)', async () => {
        mockGetCtx.mockResolvedValueOnce({
          email: 'mike@triarchsecurity.com',
          isStaff: true,
          memberships: [],
        });
        const result = await callSignIn({ email: 'mike@triarchsecurity.com' });
        expect(result).toBe(true);
      });

      it('returns true for customer admin with at least one membership', async () => {
        mockGetCtx.mockResolvedValueOnce({
          email: 'customer@example.com',
          isStaff: false,
          memberships: [{ projectKey: 'acme', role: 'admin' }],
        });
        const result = await callSignIn({ email: 'customer@example.com' });
        expect(result).toBe(true);
      });

      it('returns true for customer viewer with at least one membership', async () => {
        mockGetCtx.mockResolvedValueOnce({
          email: 'viewer@example.com',
          isStaff: false,
          memberships: [{ projectKey: 'acme', role: 'viewer' }],
        });
        const result = await callSignIn({ email: 'viewer@example.com' });
        expect(result).toBe(true);
      });

      it('returns true for staff who is also a project member', async () => {
        mockGetCtx.mockResolvedValueOnce({
          email: 'staff-member@triarchsecurity.com',
          isStaff: true,
          memberships: [{ projectKey: 'acme', role: 'admin' }],
        });
        const result = await callSignIn({ email: 'staff-member@triarchsecurity.com' });
        expect(result).toBe(true);
      });
    });
    ```
