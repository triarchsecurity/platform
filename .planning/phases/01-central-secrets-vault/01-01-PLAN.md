    # Create + populate each of 7 secrets (per locked decision: automatic replication = no --replication-policy flag, defaults to "automatic")

               GITHUB_APP_ID GITHUB_APP_PRIVATE_KEY GITHUB_APP_INSTALLATION_ID SLACK_USER_MAP; do
      # Create secret container with automatic (multi-region) replication
      gcloud secrets create "$KEY" \
        --project=triarch-vault \

    gcloud secrets list --project=triarch-vault --format="value(name)" | sort
