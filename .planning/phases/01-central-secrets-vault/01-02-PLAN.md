    **src/index.ts** — implements the package per RESEARCH.md "Complete getSecret implementation" + single-flight pattern from github-app.ts:
    ```typescript
    import { SecretManagerServiceClient } from '@google-cloud/secret-manager';

    const VAULT_PROJECT = 'triarch-vault';
    const TTL_MS = 300_000; // 300 seconds per D-06

    export class SecretNotFoundError extends Error {
      constructor(name: string) {
        super(
          `Secret '${name}' not found in vault and no fallback in process.env. ` +
          `Check vault setup at https://console.cloud.google.com/security/secret-manager?project=triarch-vault`
        );
        this.name = 'SecretNotFoundError';
      }
    }

    type CacheEntry = { value: string; expiresAt: number };

    let _client: SecretManagerServiceClient | null = null;
    const _cache = new Map<string, CacheEntry>();
    const _inflight = new Map<string, Promise<string>>();

    function client(): SecretManagerServiceClient {
      if (!_client) _client = new SecretManagerServiceClient();
      return _client;
    }

    async function fetchFromVault(name: string): Promise<string> {
      const resourceName = `projects/${VAULT_PROJECT}/secrets/${name}/versions/latest`;
      const [version] = await client().accessSecretVersion({ name: resourceName });
      const data = version.payload?.data;
      if (!data) throw new Error(`Secret '${name}' has no payload`);
      return Buffer.isBuffer(data) ? data.toString('utf8') : Buffer.from(data).toString('utf8');
    }

    export async function getSecret(name: string): Promise<string> {
      const now = Date.now();
      const hit = _cache.get(name);
      if (hit && now < hit.expiresAt) return hit.value;

      // Single-flight: collapse concurrent misses on the same key into one fetch
      const inflight = _inflight.get(name);
      if (inflight) return inflight;

      const promise = (async () => {
        try {
          const value = await fetchFromVault(name);
          _cache.set(name, { value, expiresAt: Date.now() + TTL_MS });
          return value;
        } catch {
          const envValue = process.env[name];
          if (envValue !== undefined) return envValue;
          throw new SecretNotFoundError(name);
        } finally {
          _inflight.delete(name);
        }
      })();

      _inflight.set(name, promise);
      return promise;
    }

    /** Test-only helper. Resets module-level cache + client + inflight latches. */
    export function _resetForTests(): void {
      _client = null;
      _cache.clear();
      _inflight.clear();
    }
    ```

    }

    beforeEach(() => {
      _resetForTests();
      mockAccessSecretVersion.mockReset();
      mockConstructor.mockReset();
      vi.useRealTimers();
      delete process.env.TEST_SECRET;
      delete process.env.MISSING_SECRET;
    });

    afterEach(() => {
      delete process.env.TEST_SECRET;
      delete process.env.MISSING_SECRET;
    });

    describe('getSecret — cache', () => {
      it('returns cached value within TTL (1 fetch for 2 calls)', async () => {
        mockAccessSecretVersion.mockResolvedValueOnce(payload('hello'));

        const b = await getSecret('TEST_SECRET');
        expect(a).toBe('hello');
        expect(b).toBe('hello');
        expect(mockAccessSecretVersion).toHaveBeenCalledTimes(1);
      });

        mockAccessSecretVersion.mockResolvedValueOnce(payload('v1')).mockResolvedValueOnce(payload('v2'));
        expect(await getSecret('TEST_SECRET')).toBe('v1');
        vi.advanceTimersByTime(301_000);

      it('isolates cache by key (no collision)', async () => {
        mockAccessSecretVersion
          .mockResolvedValueOnce(payload('a-val'))
          .mockResolvedValueOnce(payload('b-val'));
        expect(await getSecret('A')).toBe('a-val');
        expect(await getSecret('B')).toBe('b-val');
      });

    describe('getSecret — vault success', () => {

        mockAccessSecretVersion.mockResolvedValueOnce(payload('xoxb-test-token'));
