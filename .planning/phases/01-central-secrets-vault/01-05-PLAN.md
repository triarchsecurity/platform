// Lines 19-34 — getSlackClient
async function getSlackClient(): Promise<WebClient> {
  if (_slack) return _slack;
  let botToken: string | undefined;
  try {
    const result = await crmQuery<{ value: string }>('SELECT value FROM settings WHERE key = $1', ['slack_bot_token']);
    if (result.rows.length > 0 && result.rows[0].value) {
      botToken = decryptKey(result.rows[0].value);
    }
  } catch {
    // DB lookup failed — fall through to env var
  }
  if (!botToken) botToken = process.env.SLACK_BOT_TOKEN;
  if (!botToken) throw new Error('Slack bot token not found in CRDB settings or SLACK_BOT_TOKEN env var');
  _slack = new WebClient(botToken);
  return _slack;
}

// Lines 37-49 — getSlackSigningSecret
export async function getSlackSigningSecret(): Promise<string> {
  try {
    const result = await crmQuery<{ value: string }>('SELECT value FROM settings WHERE key = $1', ['slack_signing_secret']);
    if (result.rows.length > 0 && result.rows[0].value) {
      return decryptKey(result.rows[0].value);
    }
  } catch {
    // DB lookup failed — fall through to env var
  }
  const secret = process.env.SLACK_SIGNING_SECRET;
  if (!secret) throw new Error('Slack signing secret not found in CRDB settings or SLACK_SIGNING_SECRET env var');
  return secret;
}
