    3. Create ProjectSubNav.tsx as a `'use client'` component. Required structure:
       ```tsx
       'use client';
       import Link from 'next/link';
       import { usePathname } from 'next/navigation';

       const TABS = [
         { label: 'Releases', segment: 'releases' },
         { label: 'Bugs', segment: 'bugs' },
         { label: 'Features', segment: 'features' },
       ] as const;

       export default function ProjectSubNav({ slug }: { slug: string }) {
         const pathname = usePathname() ?? '';
         const base = `/projects/${slug}`;
         return (
           <nav
             aria-label="Project sections"
             className="border-b border-zinc-800 bg-zinc-900/30"
           >
             <div className="max-w-6xl mx-auto px-4 sm:px-8 overflow-x-auto">
               <ul className="flex items-stretch gap-6 whitespace-nowrap">
                 {TABS.map((tab) => {
                   const href = `${base}/${tab.segment}`;
                   const active = pathname.startsWith(`${base}/${tab.segment}`);
                   return (
                     <li key={tab.segment}>
                       <Link
                         href={href}
                         aria-current={active ? 'page' : undefined}
                         className={
                           active
                             ? 'inline-block py-3 text-sm font-medium text-teal-300 border-b-2 border-teal-400'
                             : 'inline-block py-3 text-sm font-medium text-zinc-400 hover:text-zinc-200 border-b-2 border-transparent'
                         }
                       >
                         {tab.label}
                       </Link>
                     </li>
                   );
                 })}
               </ul>
             </div>
           </nav>
         );
       }
       ```

       Color palette decision (per CONTEXT.md UX-01 D-02 + UX-02 palette): active uses `text-teal-300 border-teal-400` (saturated teal-cyan, matches the StatusPill `approved` family that admin's release-logs page uses).
