    ```typescript
    /**
     * Integration test for RC-08 — concurrent multi-branch approval safety (D-16, D-17).
     *
     * Per RESEARCH.md "approveRelease Concurrency Model":
     *   approveRelease is per-row idempotent, keyed by release.id UUID.
     *   Two concurrent approvals on different branches operate on different rows —
     *   no shared lock, no cross-contamination at the database level.
     *
     * This test does NOT use a real DB. It mocks db.transaction to track per-call
     * insert/update args and asserts that each parallel call sees its own inputs only.
     *
     * End-to-end multi-branch validation against a real GitHub merge is the Phase 8
     * Truth+Treason pilot (PILOT-02).
     */
    import { describe, it, expect, beforeEach, vi } from 'vitest';

    // ── Mock: @/lib/github-app (so promoteAndAudit's dispatch call is captured) ──

    vi.mock('@/lib/github-app', () => ({
      dispatchWorkflow: vi.fn().mockResolvedValue({ ok: true, status: 204 }),
    }));

    // ── Mock: @/lib/slack (promoteAndAudit calls postSlackThreadedReply / updateSlackMessage) ──

    vi.mock('@/lib/slack', () => ({
      postSlackThreadedReply: vi.fn().mockResolvedValue({ ok: true }),
      updateSlackMessage: vi.fn().mockResolvedValue({ ok: true }),
    }));

    // ── Mock: @/lib/db with per-call capture ─────────────────────────────────────

    const txInsertCalls: Array<{ values: Record<string, unknown> }> = [];
    const txUpdateCalls: Array<{ id: string; status: string }> = [];
    const promotionUpdateCalls: Array<Record<string, unknown>> = [];
    const projectSelectMock = vi.fn();

    vi.mock('@/lib/db', () => ({
      db: {
        // approveRelease.idempotency-check: SELECT releaseApprovals → returns []
        // promoteAndAudit project lookup: SELECT projects → returns [{ githubRepo: '...' }]
        select: () => ({
          from: () => ({
            where: () => {
              return {
                orderBy: () => ({ limit: () => Promise.resolve([]) }),
                // For promoteAndAudit's project lookup (no orderBy/limit), `where` resolves directly.
                then: (onFulfilled: (rows: unknown[]) => void) =>
                  Promise.resolve(projectSelectMock()).then(onFulfilled),
              };
            },
          }),
        }),
        update: () => ({
          set: (args: Record<string, unknown>) => ({
            where: () => {
              promotionUpdateCalls.push(args);
              return Promise.resolve(undefined);
            },
          }),
        }),
        transaction: async (fn: (tx: unknown) => Promise<unknown>) => {
          const tx = {
            insert: () => ({
              values: (vals: Record<string, unknown>) => ({
                returning: () =>
                  Promise.resolve([{ id: `app-${txInsertCalls.length + 1}`, ...vals }]),
              }),
            }),
            update: () => ({
              set: (args: { status: string }) => ({
                where: () => ({
                  returning: () => {
                    // Capture the row id from the closure of the parent fn — but here we
                    // don't have direct access; instead, derive id from txUpdateCalls.length
                    // mapped to the matching insert call. Simpler: we extract the release.id
                    // from the test's outer state via the seeded releases.
                    const releaseId =
                      txInsertCalls[txInsertCalls.length - 1]?.values?.releaseId ?? 'unknown';
                    txUpdateCalls.push({ id: releaseId as string, status: args.status });
                    return Promise.resolve([{ id: releaseId, status: args.status }]);
                  },
                }),
              }),
            }),
          };
          // Each transaction first inserts then updates — capture insert here:
          const originalInsert = tx.insert.bind(tx);
          tx.insert = (...args: unknown[]) => {
            const ret = originalInsert(...(args as []));
            const originalValues = ret.values.bind(ret);
            ret.values = (vals: Record<string, unknown>) => {
              txInsertCalls.push({ values: vals });
              return originalValues(vals);
            };
            return ret;
          };
          return fn(tx);
        },
      },
    }));

    vi.mock('@/db/schema', async () => {
      const actual = await vi.importActual<typeof import('@/db/schema')>('@/db/schema');
      return {
        releaseLogs: actual.releaseLogs,
        releaseApprovals: actual.releaseApprovals,
        projects: actual.projects,
      };
    });

    // ── Imports AFTER mocks ──────────────────────────────────────────────────────

    import { approveRelease } from '@/lib/release-actions';
    import { promoteAndAudit } from '@/lib/release-promotion';
    import { dispatchWorkflow } from '@/lib/github-app';

    // ── Fixtures ─────────────────────────────────────────────────────────────────

    const RELEASE_FONT = {
      id: 'rel-font-uuid',
      project: 'truth-treason',
      version: 'v0.4.2',
      releaseType: 'patch',
      releasedAt: new Date(),
      releasedBy: null,
      summary: null,
      entries: [],
      env: 'dev',
      status: 'dev',
      commitSha: 'sha-font',
      deployedAt: new Date(),
      branch: 'feat/change-font',
      promotionDispatchedAt: null,
      promotionDispatchedBy: null,
      metadata: {},
      createdAt: new Date(),
    } as unknown as Parameters<typeof approveRelease>[0]['release'];

    const RELEASE_AUDIO = {
      ...(RELEASE_FONT as object),
      id: 'rel-audio-uuid',
      version: 'v0.5.0',
      branch: 'feat/add-audio',
      commitSha: 'sha-audio',
    } as unknown as Parameters<typeof approveRelease>[0]['release'];

    beforeEach(() => {
      vi.clearAllMocks();
      txInsertCalls.length = 0;
      txUpdateCalls.length = 0;
      promotionUpdateCalls.length = 0;
      projectSelectMock.mockReturnValue([
        { githubRepo: 'MyAlterLego/truth-treason' },
      ]);
    });

    describe('Concurrent multi-branch approval safety (RC-08)', () => {
      it('two parallel approveRelease calls on different branches both succeed independently', async () => {
        const [resultFont, resultAudio] = await Promise.all([
          approveRelease({
            release: RELEASE_FONT,
            approverEmail: 'mike@triarchsecurity.com',
            ipAddress: '10.0.0.1',
            userAgent: 'Mozilla/5.0 font-tester',
          }),
          approveRelease({
            release: RELEASE_AUDIO,
            approverEmail: 'sam@triarchsecurity.com',
            ipAddress: '10.0.0.2',
            userAgent: 'Mozilla/5.0 audio-tester',
          }),
        ]);

        // Both succeeded with fresh approvals
        expect(resultFont.ok).toBe(true);
        expect(resultAudio.ok).toBe(true);
        if (resultFont.ok) expect(resultFont.alreadyApproved).toBe(false);
        if (resultAudio.ok) expect(resultAudio.alreadyApproved).toBe(false);

        // Two distinct insert calls captured — each releaseApprovals insert has its own releaseId
        expect(txInsertCalls).toHaveLength(2);
        const insertedReleaseIds = txInsertCalls.map((c) => c.values.releaseId).sort();
        expect(insertedReleaseIds).toEqual(['rel-audio-uuid', 'rel-font-uuid']);
      });

      it('per-call inputs (approverEmail, ipAddress, userAgent) flow only into their own insert', async () => {
        await Promise.all([
          approveRelease({
            release: RELEASE_FONT,
            approverEmail: 'mike@triarchsecurity.com',
            ipAddress: '10.0.0.1',
            userAgent: 'font-ua',
          }),
          approveRelease({
            release: RELEASE_AUDIO,
            approverEmail: 'sam@triarchsecurity.com',
            ipAddress: '10.0.0.2',
            userAgent: 'audio-ua',
          }),
        ]);

        // Find each insert by releaseId and assert its specific approverEmail/ipAddress/userAgent
        const fontInsert = txInsertCalls.find((c) => c.values.releaseId === 'rel-font-uuid');
        const audioInsert = txInsertCalls.find((c) => c.values.releaseId === 'rel-audio-uuid');

        expect(fontInsert?.values.approverEmail).toBe('mike@triarchsecurity.com');
        expect(fontInsert?.values.ipAddress).toBe('10.0.0.1');
        expect(fontInsert?.values.userAgent).toBe('font-ua');

        expect(audioInsert?.values.approverEmail).toBe('sam@triarchsecurity.com');
        expect(audioInsert?.values.ipAddress).toBe('10.0.0.2');
        expect(audioInsert?.values.userAgent).toBe('audio-ua');
      });

      it('two parallel promoteAndAudit calls dispatch with each branch independently (no serialization)', async () => {
        // Build the input for each parallel promotion (mock branches already set in fixtures).
        const baseInput = (release: typeof RELEASE_FONT) => ({
          release,
          actorEmail: 'mike@triarchsecurity.com',
          channelId: 'C_RELEASE_APPROVALS',
          messageTs: `170000000.${release.id.slice(-3)}`,
          slackUserName: 'mike',
        });

        await Promise.all([
          promoteAndAudit(baseInput(RELEASE_FONT)),
          promoteAndAudit(baseInput(RELEASE_AUDIO)),
        ]);

        expect(dispatchWorkflow).toHaveBeenCalledTimes(2);
        const branches = (dispatchWorkflow as unknown as { mock: { calls: Array<[{ inputs: { branch: string } }]> } })
          .mock.calls.map((c) => c[0].inputs.branch).sort();
        expect(branches).toEqual(['feat/add-audio', 'feat/change-font']);

        // Each call also wrote its metadata (post-Plan-06-01 — captured in promotionUpdateCalls)
        expect(promotionUpdateCalls.length).toBe(2);
        // Both updates carry promotionDispatchedAt + promotionDispatchedBy
        for (const call of promotionUpdateCalls) {
          expect(call.promotionDispatchedAt).toBeInstanceOf(Date);
          expect(call.promotionDispatchedBy).toBe('mike@triarchsecurity.com');
        }
      });
    });
    ```
