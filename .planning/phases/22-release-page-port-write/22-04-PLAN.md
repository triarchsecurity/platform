       ```typescript
       export async function postReleaseApprovalNotification(input: PostReleaseApprovalInput): Promise<{ok: boolean; ts?: string; error?: string}> {
         const token = await getPortalBotToken();
         if (!token) { console.warn('[portal-slack] PORTAL_SLACK_BOT_TOKEN not set — skipping notification'); return { ok: false, error: 'no_token' }; }

         const branchDisplay = sanitizeForSlack(input.branch ?? 'main');
         const safeVersion = sanitizeForSlack(input.version);
         const safeFeedback = sanitizeForSlack(input.feedbackExcerpt ?? '');

         const blocks: unknown[] = [
           {
             type: 'section',
             text: {
               type: 'mrkdwn',
               text: `:rocket: *${branchDisplay} ${safeVersion} approved by ${input.approverEmail} (via portal)*\n*Project:* ${input.project}\n*Status:* approved`,
             },
           },
         ];
         if (safeFeedback) {
           blocks.push({
             type: 'section',
             text: {
               type: 'mrkdwn',
               text: `> ${safeFeedback}` + ((input.feedbackOverflowCount ?? 0) > 0 ? `\n_(${input.feedbackOverflowCount} more comments)_` : ''),
             },
           });
         }
         // NO action buttons — portal does not have SLACK_PAYLOAD_SECRET; staff buttons are admin-only.

         const res = await fetch('https://slack.com/api/chat.postMessage', {
           method: 'POST',
           headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' },
           body: JSON.stringify({
             channel: process.env.SLACK_RELEASE_APPROVAL_CHANNEL ?? '#release-approvals',
             text: `Release Approved (via portal): ${input.project} ${safeVersion}`,
             blocks,
           }),
         });
         const data = await res.json() as { ok: boolean; ts?: string; error?: string };
         if (!data.ok) console.warn(`[portal-slack] approval notification failed: ${data.error}`);
         return data;
       }

       // postReleaseRejectionNotification: same shape; text "Rejected by {email} (via portal)" + reason if non-null.
       ```
