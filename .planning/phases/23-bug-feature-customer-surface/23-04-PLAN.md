```typescript
export async function POST(req: NextRequest, { params }: { params: Promise<{ slug: string }> }) {
  // 1. Auth
  const session = await getPortalSession();
  if (!session?.user?.email) return NextResponse.json({ error: 'unauthorized' }, { status: 401 });
  const ctx = await getCurrentUserContext({ user: { email: session.user.email } });
  if (!ctx) return NextResponse.json({ error: 'unauthorized' }, { status: 401 });

  // Pitfall 2 — email length guard (varchar(128) column)
  if (ctx.email.length > 128) return NextResponse.json({ error: 'invalid_email' }, { status: 400 });

  const { slug } = await params;

  // 2. Project + membership
  const [project] = await db.select({ key: projects.key, name: projects.name }).from(projects).where(eq(projects.key, slug));
  if (!project) return NextResponse.json({ error: 'Not found' }, { status: 404 });
  const membership = ctx.memberships.find((m) => m.project_key === project.key);
  if (!ctx.isStaff && !membership) return NextResponse.json({ error: 'Not found' }, { status: 404 });

  // 3. Body validation
  const body = await req.json().catch(() => null);
  if (!body || typeof body !== 'object') return NextResponse.json({ error: 'invalid_body' }, { status: 400 });
  const { title, description, severity, stepsToReproduce, expectedBehavior, actualBehavior } = body as Record<string, unknown>;
  if (typeof title !== 'string' || !title.trim()) return NextResponse.json({ error: 'title_required' }, { status: 400 });
  if (typeof description !== 'string' || !description.trim()) return NextResponse.json({ error: 'description_required' }, { status: 400 });

  const TITLE_MAX = 256;
  if (title.length > TITLE_MAX) return NextResponse.json({ error: 'title_too_long' }, { status: 400 });

  // 4. INSERT — schema defaults handle severity/priority/status if not provided
  const [bug] = await db.insert(bugReports).values({
    project: project.key,
    reportedByUserId: ctx.email,
    reportedByEmail: ctx.email,
    reportedByName: session.user.name ?? null,
    title: title.trim(),
    description: description.trim(),
    severity: typeof severity === 'string' && severity.trim() ? severity : 'medium',
    stepsToReproduce: typeof stepsToReproduce === 'string' && stepsToReproduce.trim() ? stepsToReproduce.trim() : null,
    expectedBehavior: typeof expectedBehavior === 'string' && expectedBehavior.trim() ? expectedBehavior.trim() : null,
    actualBehavior: typeof actualBehavior === 'string' && actualBehavior.trim() ? actualBehavior.trim() : null,
  }).returning();

  // 5. Slack post FIRST (3-sec budget; fire-and-forget; failure does NOT roll back)
  let slackTs: string | undefined;
  let slackChannel: string | undefined;
  try {
    const slackResult = await postBugSubmissionNotification({
      bug, projectName: project.name, reporterEmail: ctx.email,
    });
    if (slackResult.ok) { slackTs = slackResult.ts; slackChannel = slackResult.channel; }
    else { console.warn('[portal-bugs] Slack notification failed', { bugId: bug.id, error: slackResult.error }); }
  } catch (err) {
    console.warn('[portal-bugs] Slack notification threw', { bugId: bug.id, error: String(err) });
  }

  // 6. Best-effort UPDATE slack_message_ts + slack_channel_id (RESEARCH.md OQ-2)
  if (slackTs) {
    try {
      await db.update(bugReports)
        .set({ slackMessageTs: slackTs, slackChannelId: slackChannel ?? null })
        .where(eq(bugReports.id, bug.id));
    } catch (err) {
      console.warn('[portal-bugs] slack_message_ts update failed', { bugId: bug.id, error: String(err) });
    }
  }

  // 7. Response — 201 with bug.id so client can redirect
  return NextResponse.json({ ok: true, bug }, { status: 201 });
}
```

         const safeTitle = sanitizeForSlack(input.bug.title);

       import { useState } from 'react';

       export function BugForm({ projectSlug }: { projectSlug: string }) {
         const router = useRouter();
         const [title, setTitle] = useState('');
         const [description, setDescription] = useState('');
         const [severity, setSeverity] = useState('medium');
         const [stepsToReproduce, setStepsToReproduce] = useState('');
         const [expectedBehavior, setExpectedBehavior] = useState('');
         const [actualBehavior, setActualBehavior] = useState('');
         const [submitting, setSubmitting] = useState(false);
         const [error, setError] = useState<string | null>(null);

         const canSubmit = title.trim().length > 0 && description.trim().length > 0 && !submitting;

         async function handleSubmit(e: React.FormEvent) {
           e.preventDefault();
           if (!canSubmit) return;
           setSubmitting(true);
           setError(null);
           try {
             const res = await fetch(`/api/projects/${projectSlug}/bugs`, {
               method: 'POST',
               headers: { 'Content-Type': 'application/json' },
               body: JSON.stringify({
                 title,
                 description,
                 severity,
                 stepsToReproduce: stepsToReproduce.trim() || undefined,
                 expectedBehavior: expectedBehavior.trim() || undefined,
                 actualBehavior: actualBehavior.trim() || undefined,
               }),
             });
             const json = await res.json().catch(() => ({}));
             if (!res.ok) {
               setError(typeof json.error === 'string' ? json.error : 'Submission failed');
               setSubmitting(false);
               return;
             }
             const bugId = json.bug?.id;
             if (!bugId) {
               setError('Unexpected response — bug id missing');
               setSubmitting(false);
               return;
             }
             router.push(`/projects/${projectSlug}/bugs/${bugId}`);
           } catch (err) {
             setError(`Network error: ${String(err)}`);
             setSubmitting(false);
           }

                 id="title"
                 type="text"
                 value={title}
                 onChange={(e) => setTitle(e.target.value)}
                 maxLength={256}
                 required
                 className="w-full rounded bg-zinc-900 border border-zinc-700 p-2 text-zinc-100"
               />
             </div>

             <div>
               <label htmlFor="description" className="block text-sm font-medium text-zinc-300 mb-1">Description <span className="text-red-400">*</span></label>
               <textarea
                 id="description"
                 value={description}
                 onChange={(e) => setDescription(e.target.value)}

                 required

               />
             </div>

             <div>
               <label htmlFor="severity" className="block text-sm font-medium text-zinc-300 mb-1">Severity</label>
               <select
                 id="severity"
                 value={severity}
                 onChange={(e) => setSeverity(e.target.value)}
                 className="rounded bg-zinc-900 border border-zinc-700 p-2 text-zinc-100"
               >
                 <option value="low">Low</option>
                 <option value="medium">Medium</option>
                 <option value="high">High</option>
                 <option value="critical">Critical</option>
               </select>
             </div>

             <details className="rounded border border-zinc-800 bg-zinc-900/50 p-3">
               <summary className="cursor-pointer text-sm text-zinc-400">Optional details (steps to reproduce, expected, actual)</summary>
