       ```typescript
       import { notFound, redirect } from 'next/navigation';
       import Link from 'next/link';
       import { eq, and, desc } from 'drizzle-orm';
       import { getCurrentUserContext } from '@myalterlego/triarch-shared/auth';
       import { projects, featureRequests } from '@myalterlego/triarch-shared/schema';
       import { db } from '@/lib/db';
       import { getPortalSession } from '@/lib/session';
       import CustomerHeader from '@/app/projects/CustomerHeader';
       import { FeatureListClient } from './FeatureListClient';

       const PAGE_SIZE = 20;

       const ALLOWED_STATUSES = new Set([
         'submitted', 'triaged', 'plan_generated', 'approved', 'in_progress',
         'shipped', 'closed', 'declined', 'deferred',
       ]);

       export default async function FeatureListPage({
         params,
         searchParams,
       }: {
         params: Promise<{ slug: string }>;
         searchParams: Promise<{ status?: string; project?: string }>;
       }) {
         const session = await getPortalSession();
         if (!session?.user?.email) redirect('/login');
         const ctx = await getCurrentUserContext({ user: { email: session.user.email } });
         const { slug } = await params;
         const sp = await searchParams;

         // Pitfall 4 guard — sp.project deliberately ignored
         const statusFilter = sp.status && ALLOWED_STATUSES.has(sp.status) ? sp.status : null;

         const [project] = await db
           .select({ key: projects.key, name: projects.name })
           .from(projects)
           .where(eq(projects.key, slug));
         if (!project) notFound();

         const membership = ctx?.memberships.find((m) => m.project_key === project.key);
         if (!ctx || (!ctx.isStaff && !membership)) notFound();

         const where = statusFilter
           ? and(eq(featureRequests.project, project.key), eq(featureRequests.status, statusFilter))
           : eq(featureRequests.project, project.key);

         const rows = await db
           .select()
           .from(featureRequests)
           .where(where)
           .orderBy(desc(featureRequests.createdAt))
           .limit(PAGE_SIZE + 1);

         const hasMore = rows.length > PAGE_SIZE;
         const pageRows = hasMore ? rows.slice(0, PAGE_SIZE) : rows;

         return (
           <main className="min-h-screen bg-zinc-950 text-zinc-100">
             <CustomerHeader projectName={project.name} />
             <div className="max-w-5xl mx-auto px-4 py-6 space-y-6">
               <div className="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
                 <h1 className="text-2xl font-bold">Features</h1>
                 <Link
                   href={`/projects/${project.key}/features/new`}
                   className="px-4 py-2 rounded bg-violet-600 hover:bg-violet-500 text-white text-sm font-medium text-center sm:text-left"
                 >
                   Submit a feature request
                 </Link>
               </div>
               <FeatureListClient
                 projectSlug={project.key}
                 features={pageRows}
                 hasMore={hasMore}
                 statusFilter={statusFilter}
               />
             </div>
           </main>
         );
       }
       ```

    2. Create `portal/src/app/projects/[slug]/features/FeatureListClient.tsx` — client island (mirror of BugListClient with FeatureStatusPill + 9-status options):
       ```typescript
       'use client';
       import Link from 'next/link';
       import { useRouter, usePathname } from 'next/navigation';
       import { FeatureStatusPill } from '@/components/StatusPill';

       const STATUS_OPTIONS: Array<{ value: string | null; label: string }> = [
         { value: null, label: 'All' },
         { value: 'submitted', label: 'Submitted' },
         { value: 'triaged', label: 'Triaged' },
         { value: 'plan_generated', label: 'Plan Generated' },
         { value: 'approved', label: 'Approved' },
         { value: 'in_progress', label: 'In Progress' },
         { value: 'shipped', label: 'Shipped' },
         { value: 'closed', label: 'Closed' },
         { value: 'declined', label: 'Declined' },
         { value: 'deferred', label: 'Deferred' },
       ];

       export type FeatureListRow = {
         id: string;
         title: string;
         status: string;
         priority: string | null;
         createdAt: Date | string;
         requestedByEmail: string | null;
         upvotes: number | null;
       };

       export function FeatureListClient({
         projectSlug,
         features,
         hasMore,
         statusFilter,
       }: {
         projectSlug: string;
         features: FeatureListRow[];
         hasMore: boolean;
         statusFilter: string | null;
       }) {
         const router = useRouter();
         const pathname = usePathname();

         const handleChipClick = (value: string | null) => {
           const params = new URLSearchParams();
           if (value) params.set('status', value);
           const qs = params.toString();
           router.replace(qs ? `${pathname}?${qs}` : pathname, { scroll: false });
         };

         if (features.length === 0 && !statusFilter) {
           return (
             <div className="rounded-lg border border-zinc-800 bg-zinc-900 p-8 text-center text-zinc-400">
               No features yet.
             </div>
           );
         }

         return (
           <div className="space-y-4">
             <div className="flex flex-wrap gap-2">
               {STATUS_OPTIONS.map((opt) => {
                 const isActive = statusFilter === opt.value || (!statusFilter && opt.value === null);
                 return (
                   <button
                     key={opt.label}
                     type="button"
                     onClick={() => handleChipClick(opt.value)}
                     className={`px-3 py-1 rounded text-xs ${
                       isActive ? 'bg-violet-600 text-white' : 'bg-zinc-800 text-zinc-400 hover:bg-zinc-700'
                     }`}
                   >
                     {opt.label}
                   </button>
                 );
               })}
             </div>

             <div className="rounded-lg border border-zinc-800 divide-y divide-zinc-800">
               {features.length === 0 ? (
                 <div className="p-6 text-center text-zinc-500">No features match this filter.</div>
               ) : (
                 features.map((feat) => (
                   <Link
                     key={feat.id}
                     href={`/projects/${projectSlug}/features/${feat.id}`}
                     className="block p-4 hover:bg-zinc-900 transition-colors"
                   >
                     <div className="flex items-start justify-between gap-3">
                       <div className="flex-1 min-w-0">
                         <h3 className="font-medium text-zinc-100 truncate">{feat.title}</h3>
                         <p className="text-xs text-zinc-500 mt-1">
                           {feat.requestedByEmail ?? 'Unknown reporter'} · {new Date(feat.createdAt).toLocaleDateString()}
                         </p>
                       </div>
                       <div className="flex-shrink-0">
                         <FeatureStatusPill status={feat.status} />
                       </div>
                     </div>
                   </Link>
                 ))
               )}
             </div>

             {hasMore ? (
               <div className="text-center text-zinc-500 text-sm">
                 Showing first 20 results — older features not yet paginable.
               </div>
             ) : null}
           </div>
         );
       }
       ```

       NOTE: `upvotes` field exists in the row type but is not rendered in v2.2 — per CONTEXT.md "Customer upvote on feature requests deferred". Including the field on the row type makes future renders trivial without a schema reshape.
