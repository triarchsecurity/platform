File body to paste:

```yaml
# shared-workflows/.github/workflows/db-migrate.yml
# Reusable workflow: run drizzle-kit push against the dev or prod database.
# Replaces the manual ritual:
#   firebase apphosting:secrets:access DATABASE_URL --project X | psql -f migration.sql
#
# Triggered via workflow_call from consumer ci-cd.yml OR workflow_dispatch (manual one-shot).
# CONTEXT D-19 + Specifics: manual workflow_dispatch only for v2.0; auto-trigger on schema.ts diff is v3.

name: DB Migrate

on:
  workflow_call:
    inputs:
      environment:
        description: 'dev or prod'
        required: false
        type: string
        default: 'dev'
      firebase_project_id:
        description: 'Firebase project ID to read the database secret from.'
        required: true
        type: string
      consumer_repo:
        description: 'Consumer repo (owner/repo) to checkout for drizzle config.'
        required: true
        type: string
      consumer_ref:
        description: 'Git ref to checkout. Defaults to environment branch (dev for dev, main for prod).'
        required: false
        type: string
        default: ''
    secrets:
      FIREBASE_SA_KEY:
        description: 'GCP service-account JSON with Firebase secret access.'
        required: true
      GH_PAT:
        description: 'PAT to checkout consumer_repo if private. Falls back to github.token.'
        required: false
  workflow_dispatch:
    inputs:
      environment:
        description: 'dev or prod'
        required: false
        type: string
        default: 'dev'
      firebase_project_id:
        description: 'Firebase project ID (e.g. triarch-dev-website for admin dev)'
        required: true
        type: string
      consumer_repo:
        description: 'Consumer repo (e.g. MyAlterLego/triarch-dev)'
        required: true
        type: string
      consumer_ref:
        description: 'Git ref to checkout (leave blank for default)'
        required: false
        type: string
        default: ''

permissions:
  contents: read
  packages: read

jobs:
  migrate:
    name: drizzle-kit push (${{ inputs.environment }})
    runs-on: ubuntu-latest
    timeout-minutes: 10

    steps:
      - name: Checkout consumer repo
        uses: actions/checkout@v4
        with:
          repository: ${{ inputs.consumer_repo }}
          token: ${{ secrets.GH_PAT || github.token }}
          ref: ${{ inputs.consumer_ref != '' && inputs.consumer_ref || (inputs.environment == 'dev' && 'dev' || 'main') }}

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'
          registry-url: 'https://npm.pkg.github.com'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci
        env:
          NODE_AUTH_TOKEN: ${{ secrets.GH_PAT || github.token }}

      - name: Install Firebase CLI
        run: npm install -g firebase-tools

      - name: Authenticate to Google Cloud
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.FIREBASE_SA_KEY }}

      - name: Set up Cloud SDK
        uses: google-github-actions/setup-gcloud@v2

      - name: Read database secret
        id: read_secret
        env:
          ENVIRONMENT: ${{ inputs.environment }}
          FIREBASE_PROJECT_ID: ${{ inputs.firebase_project_id }}
        run: |
          if [ "$ENVIRONMENT" = "dev" ]; then
            SECRET_NAME="DATABASE_URL_DEV"
          else
            SECRET_NAME="DATABASE_URL"
          fi
          echo "Reading $SECRET_NAME from project $FIREBASE_PROJECT_ID"
          # Pitfall 7: trim trailing newline — single-line CRDB URI must not end with \n
          DB_URL=$(firebase apphosting:secrets:access "$SECRET_NAME" --project "$FIREBASE_PROJECT_ID" | tr -d '\n')
          # Mask the value so it cannot leak via subsequent log lines
          echo "::add-mask::$DB_URL"
          echo "database_url=$DB_URL" >> "$GITHUB_OUTPUT"

      - name: Run drizzle-kit push
        env:
          DATABASE_URL: ${{ steps.read_secret.outputs.database_url }}
          ENVIRONMENT: ${{ inputs.environment }}
        run: |
          echo "Running drizzle-kit push against $ENVIRONMENT database"
          npx drizzle-kit push
          echo "drizzle-kit push completed successfully"

      - name: Migration summary
        if: always()
        env:
          ENVIRONMENT: ${{ inputs.environment }}
          FIREBASE_PROJECT_ID: ${{ inputs.firebase_project_id }}
          JOB_STATUS: ${{ job.status }}
        run: |
          {
            echo "## DB Migration"
            echo ""
            echo "| Field | Value |"
            echo "|-------|-------|"
            echo "| Environment | \`$ENVIRONMENT\` |"
            echo "| Firebase Project | \`$FIREBASE_PROJECT_ID\` |"
            if [ "$JOB_STATUS" = "success" ]; then
              echo "| Status | SUCCESS |"
            else
              echo "| Status | FAILED |"
            fi
          } >> "$GITHUB_STEP_SUMMARY"
```

The RESEARCH draft used inline `${{ job.status }}` in run: (a deviation from project convention). Apply the env-block fix as shown above.
