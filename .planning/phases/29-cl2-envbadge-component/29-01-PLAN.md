    Step 1 — Create `src/components/EnvBadge/index.tsx`:

    ```tsx
    import React from 'react';

    export interface EnvBadgeProps {
      /** Value of NEXT_PUBLIC_ENV. Renders only when 'dev' or 'staging' (case-insensitive). */
      env?: string;
    }

    /**
     * Persistent environment badge — CL-2 of the Dev/Prod Distinction Contract.
     *
     * Renders a fixed-position pill in the bottom-right corner showing the env name
     * ("DEV" or "STAGING") when env is one of those values (case-insensitive).
     * Returns null for prod or any other value — invisible in prod chrome.
     *
     * Emits data-env="dev" (or data-env="staging") for the CL-1..6 compliance
     * scan in admin/modules/ci-cd, which HTML-parses dev URLs to assert mount
     * presence. The data-env attribute MUST be on a server-rendered element
     * (this component is a client island via the tsup "use client" banner, but
     * it renders synchronously in the initial HTML — no lazy load).
     */
    export function EnvBadge({ env }: EnvBadgeProps) {
      const normalized = env?.toLowerCase();
      if (normalized !== 'dev' && normalized !== 'staging') {
        return null;
      }

      const isDev = normalized === 'dev';
      const label = isDev ? 'DEV' : 'STAGING';
      const bgColor = isDev ? '#facc15' : '#fb923c'; // yellow-400 / orange-400
      const textColor = '#1f2937'; // zinc-800

      return (
        <div
          data-env={normalized}
          role="status"
          aria-label={`Environment: ${normalized}`}
          style={{
            position: 'fixed',
            bottom: '12px',
            right: '12px',
            zIndex: 9000, // above app content; below typical modal overlays (10000+)
            padding: '4px 10px',
            borderRadius: '9999px',
            backgroundColor: bgColor,
            color: textColor,
            fontSize: '11px',
            fontWeight: 700,
            letterSpacing: '0.05em',
            fontFamily: 'system-ui, -apple-system, sans-serif',
            boxShadow: '0 2px 8px rgba(0, 0, 0, 0.25)',
            pointerEvents: 'none',
            userSelect: 'none',
          }}
        >
          {label}
        </div>
      );
    }
    ```
